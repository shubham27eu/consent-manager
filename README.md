import matplotlib.pyplot as plt

def plot_chart_for_topK(x, kpi_arr1, kpi_arr2, kpi_arr3, kpi_arr4, kpi_arr5):
    # Sample data
    # Create a line chart
    plt.figure(figsize=(8, 6))
    # plt.plot(x, y, marker='o', linestyle='-')

    df = pd.DataFrame({
        'x_values': x,
        'y1_values': kpi_arr1,
        'y2_values': kpi_arr2,
        'y3_values': kpi_arr3,
        'y4_values': kpi_arr4,
        'y5_values': kpi_arr5
        }
    )

    # plt.plot(x, kpi_arr1, marker='o', linestyle='-')

    plt.plot(
        'x_values', 'y1_values', data=df,
        marker='o', # marker type
        markerfacecolor='blue', # color of marker
        markersize=8, # size of marker
        color='skyblue', # color of line
        linewidth=2, # change width of line
        label="path_traversal" # label for legend
    )

    plt.plot(
        'x_values', 'y2_values', data=df,
        marker='', # no marker
        color='hotpink', # color of line
        linewidth=3, # change width of line
        linestyle='dotted',
        label="path_traversal_gma" # label for legend
    )

    plt.plot(
        'x_values', 'y3_values', data=df,
        marker='', # no marker
        color='darkred', # color of line
        linewidth=2, # change width of line
        linestyle='dashed', # change type of line
        label="cost_evaluation" # label for legend
    )
    
    plt.plot(
        'x_values', 'y4_values', data=df,
        marker='', # no marker
        color='darkgreen', # color of line
        linewidth=2, # change width of line
        linestyle='dotted', # change type of line
        label="fpm" # label for legend
    )
    
    plt.plot(
        'x_values', 'y5_values', data=df,
        marker='x', # no marker
        color='orange', # color of line
        linewidth=2, # change width of line
        linestyle='dashdot', # change type of line
        label="pmi" # label for legend
    )

    # Add title and labels
    plt.title('KPI (%) w.r.t. topK RCA Nodes')
    plt.xlabel('K (top nodes in result)')
    plt.ylabel('% of path with atleast 1 rc node in top K')

    # Display grid
    # plt.grid(True)

    # show legend
    plt.legend()

    # show graph
    plt.show()
